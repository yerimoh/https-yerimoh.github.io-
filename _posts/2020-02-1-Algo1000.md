## 전체 코드
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxDepth(self, root: TreeNode) -> int:
        if root is None:
            return 0
        q = collection.deque([root])
        depth = 0

        while q:
            depth+=1
            for _ in range(len(queue)):
                cheak = q.popleft()
                if cheak.left:
                    q.append(cheak.left)
                if cheak.right:
                    q.append(cheak.right)
        
        return depth
```
